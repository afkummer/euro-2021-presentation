\frame{
   \frametitle{Take-out}

   \textbf{F\&O matheuristic}
   \begin{itemize}
      \item Flexible
      \item Time-consuming
      \item Requires a good MIP solver
   \end{itemize}

   \vspace*{18pt}

   \textbf{BRKGA}
   \begin{itemize}
      \item Also flexible
      \item Relatively fast
      \item Intensification components are important
   \end{itemize}
}
