\frame{
   \frametitle{Take-out}

   \textbf{F\&O matheuristic}
   \begin{itemize}
      \item Flexible
      \item Works well on instances up to 75 patients
      \item Time-consuming
      \item Requires a good MIP solver
   \end{itemize}

   \pause
   \vspace*{6pt}

   \textbf{BRKGA}
   \begin{itemize}
      \item Also flexible
      \item Interesting on instances with 25+ patients
      \item Relatively fast
      \item Intensification components are effective
   \end{itemize}

   \pause
   \vspace*{6pt}

   \textbf{Route interdependencies are hard to deal with!}
